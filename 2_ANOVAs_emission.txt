# Initiate libraries we will need for the analysis
library(tidyverse); library(reshape2)

# Set working directory (where documents are stored)
#setwd("C:/Users/bhc928/Desktop/Adriana")

# Read in 2018 and 2019 data files for merging
df_2018 <- read.table("2018_Data.txt", header=TRUE, sep="\t")
df_2019 <- read.table("2019_Data.txt", header=TRUE, sep="\t")

#####################EXCLUDED TWO SAMPLES
# Merge 2018 and 2019 data files (ignore "Year" column) and remove all compounds that should be excluded
# Note: here, I also exclude one sample (11-n-JUN), which exhibited overloaded peaks and dominates the results
df_merge <- full_join(df_2018[,-45], df_2019[,-7]) %>% filter(Exclude!="Y") %>% select(-Exclude, -"X10.11_n_JUN.CDF", -"X8.8_h_JUN.CDF"  ) 

View(df_merge)

# Test for duplicates (labelling errors between datasets)
df_merge %>% group_by(VOC) %>% filter(n()>1) %>% summarize(n=n())

# Convert files to long format for merging with sample information
df_melt <- melt(df_merge, id.vars=c("RRT","Group","Class","MF","VOC"), variable.name="Filename", value.name="Mass_ng")

# Load in sample information to bind with the data for plotting
df_info <- read.table("Sample_information_ALL.txt", header=TRUE, sep="\t")

# Revise Filenames so they are consistent for merging
df_melt$Filename <- gsub("X", "", df_melt$Filename, fixed=TRUE) # Remove "X" from start of filename
df_info$Filename <- gsub("-", ".", df_info$Filename, fixed=TRUE) # Replace - with .

# Merge sample information and data
df_final <- merge(df_melt, df_info, by="Filename", sort=TRUE)

#######################################################################################

# Calculate BVOC concentrations (ug L-1) = mVOC/(t*Q) and emission rates (ug m-2 h-1)
# mVOC we know, t = 10 min = 0.167 h, Q = 200 mL/min = 12 L/h
# SA = 317.8 cm2 = 0.03178 m2, tave = 5 min = 0.083 h

df_final <- df_final %>% mutate(Conc_ug_L=(Mass_ng/1000)/(0.167*12),
                                ER_ug_m2_h=(Conc_ug_L/0.083)*((V_total_cm3/1000)/0.03178))

#######################################################################################

# Blank subtraction
# First, separate samples from blanks
blanks <- df_final %>% filter(Treatment=="blank") %>% select(Site, Year, Month, VOC, RRT, Conc_blank=Conc_ug_L, ER_blank=ER_ug_m2_h)
samples <- df_final %>% filter(Treatment!="blank")

# Merge based on Year, Month, VOC, and RRT and remove duplicates
merge <- left_join(samples, blanks) %>% distinct()

# Calculate blank-subtracted emission rates (accounts for the difference in total volume)
merge <- merge %>% mutate(ER_final=ER_ug_m2_h-ER_blank)

# Convert negative values to 0
indx=which(merge$ER_final<0); merge$ER_final[indx] <- 0

# Export merged dataset
write.csv(merge, "20201026_Merged 2018_2019 dataset.csv", row.names=FALSE) ##modified manually changed September 2019 to August 2019

#######################################################################################

# Summarise emission rates by sample and group (summed emission rates)
merge$Year <- as.character(merge$Year)
total_sum <- merge %>% group_by(Site, Year, Month, Treatment, Sample) %>% summarise(N=sum(is.na(ER_final)), Total_ER=sum(ER_final, na.rm=TRUE))

class_sum <- merge %>% group_by(Site, Year, Month, Treatment, Sample, Class) %>% summarise(n=sum(is.na(ER_final)), Class_ER=sum(ER_final, na.rm=TRUE))

sample_summary <- merge(total_sum, class_sum) %>% mutate(Percent=Class_ER/Total_ER*100)

# Summarise emission rates by treatment (averaged emission rates)
total_ave <- total_sum %>% group_by(Site, Year, Month, Treatment) %>% summarise(n=n(), N_ave=mean(N, na.rm=TRUE), Total_ER_ave=mean(Total_ER, na.rm=TRUE), Total_ER_sd=sd(Total_ER, na.rm=TRUE), Total_ER_se=Total_ER_sd/sqrt(n))
class_ave <- sample_summary %>% group_by(Site, Year, Month, Treatment, Class) %>% summarise(n_group=n(), n_ave=mean(n, na.rm=TRUE), Class_ER_ave=mean(Class_ER, na.rm=TRUE), Class_ER_sd=sd(Class_ER, na.rm=TRUE), Class_ER_se=Class_ER_sd/sqrt(n_group), Percent_ave=mean(Percent, na.rm=TRUE),Percent_sd=sd(Percent, na.rm=TRUE), Percent_se=Percent_sd/sqrt(n_group))
treatment_summary <- merge(total_ave, class_ave)

View(treatment_summary)

# Create a revised treatment column for ease of plotting
treatment_summary <- treatment_summary %>% mutate(Treatment2=ifelse(Treatment=="control", "C", ifelse(Treatment=="heated", "H", ifelse(Treatment=="nitrogen", "N",ifelse(Treatment=="heated_nitrogen", "HxN", "N50")))))

# Export treatment summary data
write.csv(treatment_summary, "20201027_Merged 2018_2019 treatment summary.csv", row.names=FALSE)
